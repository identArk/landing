Get Started with IdentArk

Integrate secure credential management into your AI agents in minutes.

Your API Key

Use this key to authenticate your agents with IdentArk Cloud.

1. Install the SDK

# Install via pip
pip install identark
// Install via npm
npm install @identark/sdk

2. Initialize IdentArk

from identark import IdentArk

cs = IdentArk(
    api_key="cs_live_7x9k2m4n5p8q1v3w6y0z",
    agent_id="research-agent"
)
import { IdentArk } from '@identark/sdk';

const cs = new IdentArk({
  apiKey: 'cs_live_7x9k2m4n5p8q1v3w6y0z',
  agentId: 'research-agent'
});

3. Request Credentials

# Request a credential with specific scopes
with cs.session("github-api", scopes=["repo:read"]) as session:
    # Credential is automatically injected
    token = session.get("GITHUB_TOKEN")

    # Use the token in your API calls
    response = requests.get(
        "https://api.github.com/user/repos",
        headers={"Authorization": f"Bearer {token}"}
    )

# Credential is automatically revoked when session ends
// Request a credential with specific scopes
const session = await cs.session('github-api', {
  scopes: ['repo:read']
});

try {
  // Credential is automatically injected
  const token = session.get('GITHUB_TOKEN');

  // Use the token in your API calls
  const response = await fetch('https://api.github.com/user/repos', {
    headers: { Authorization: `Bearer ${token}` }
  });
} finally {
  // Always close the session
  await session.close();
}
